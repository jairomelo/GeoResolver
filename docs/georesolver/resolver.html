 23class BaseQuery:
 24    """
 25    Base class for geolocation API services.
 26    Handles caching, rate limiting, and basic GET requests.
 27    """
 28
 29    def __init__(
 30        self,
 31        base_url: str,
 32        cache_name: str = "geo_cache",
 33        cache_expiry: int = 86400,  # 1 day
 34        rate_limit: tuple = (30, 1),  # 30 calls per 1 second
 35        enable_cache: bool = True,
 36        verbose: bool = False
 37    ):
 38        self.base_url = base_url.rstrip("/")
 39        self.calls, self.period = rate_limit
 40        self.logger = setup_logger(self.__class__.__name__, verbose)
 41
 42        if enable_cache:
 43            requests_cache.install_cache(cache_name, expire_after=cache_expiry)
 44            self.logger.info(f"Installed cache '{cache_name}' (expires after {cache_expiry}s)")
 45
 46    @sleep_and_retry
 47    @limits(calls=30, period=1)
 48    def _limited_get(self, url: str, params: Optional[Dict[str, Any]] = None) -> requests.Response:
 49        """
 50        Internal method to perform a GET request with rate limiting.
 51        """
 52        full_url = f"{self.base_url}{url}" if not url.startswith("http") else url
 53        try:
 54            response = requests.get(full_url, params=params)
 55            response.raise_for_status()
 56            if getattr(response, "from_cache", False):
 57                self.logger.info(f"[CACHE HIT] {response.url}")
 58            else:
 59                self.logger.info(f"[API CALL] {response.url}")
 60            return response
 61        except requests.RequestException as e:
 62            self.logger.error(f"Request failed for URL: {full_url}, params: {params}, error: {e}")
 63            raise

 76class TGNQuery(BaseQuery):
 77    """
 78    A class to interact with the Getty Thesaurus of Geographic Names (TGN) SPARQL endpoint.
 79    
 80    This class provides methods to search and retrieve geographic coordinates for places
 81    using the Getty TGN linked open data service. It supports fuzzy matching of place names
 82    and filtering by country and place type.
 83
 84    Attributes:
 85        sparql (SPARQLWrapper): SPARQL endpoint wrapper instance for TGN queries
 86        lang (str): Language code for the place type (default: "en")
 87
 88    Example:
 89        >>> tgn = TGNQuery("http://vocab.getty.edu/sparql")
 90        >>> results = tgn.places_by_name("Madrid", "Spain", "ciudad")
 91        >>> coordinates = tgn.get_best_match(results, "Madrid")
 92    """
 93    def __init__(self,  lang: str = "en"):
 94        super().__init__(base_url=config["apis"]["tgn_endpoint"])
 95        self.sparql = SPARQLWrapper(self.base_url)
 96        self.sparql.setReturnFormat(JSON)
 97        self.lang = lang
 98
 99    @sleep_and_retry
100    @limits(calls=10, period=1)
101    def places_by_name(self, place_name: str, country_code: str, place_type: Union[str, None] = None) -> Union[dict, list]:
102        """
103        Search for places using the TGN SPARQL endpoint.
104        
105        Parameters:
106            place_name (str): Name of the place to search for
107            country_code (str): Country code or name
108            place_type (str): Optional type of place (e.g., 'ciudad', 'pueblo')
109        """
110
111
112        type_filter = f'?p gvp:placeType [rdfs:label "{place_type}"@{self.lang}].' if place_type else ''
113
114        query = f"""
115            PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
116            PREFIX luc: <http://www.ontotext.com/owlim/lucene#>
117            PREFIX gvp: <http://vocab.getty.edu/ontology#>
118            PREFIX xl: <http://www.w3.org/2008/05/skos-xl#>
119            PREFIX tgn: <http://vocab.getty.edu/tgn/>
120            PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
121
122            SELECT * {{
123                ?p skos:inScheme tgn:; luc:term "{place_name}"; gvp:prefLabelGVP [xl:literalForm ?pLab].
124                ?pp1 skos:inScheme tgn:; luc:term "{country_code}"; gvp:prefLabelGVP [xl:literalForm ?pp1Lab].
125                ?p gvp:broaderPartitiveExtended ?pp1.
126                {type_filter}
127            }}
128        """
129        
130        try:
131            self.sparql.setQuery(query)
132            results = self.sparql.query().convert()
133            if isinstance(results, dict) and "results" in results and "bindings" in results["results"]:
134                return results["results"]["bindings"]
135            else:
136                self.logger.error(f"Unexpected SPARQL result format for '{place_name}': {results}")
137                return []
138        except Exception as e:
139            self.logger.error(f"Error querying TGN for '{place_name}': {str(e)}")
140            return []
141
142    def get_coordinates_lod_json(self, tgn_uri: str) -> tuple:
143        json_url = tgn_uri + ".json"
144        try:
145            response = self._limited_get(json_url) 
146            data = response.json()
147            for item in data.get("identified_by", []):
148                if item.get("type") == "crm:E47_Spatial_Coordinates":
149                    coords = ast.literal_eval(item.get("value"))
150                    if isinstance(coords, list) and len(coords) == 2:
151                        lon, lat = coords
152                        return lat, lon
153            return (None, None)
154        except Exception as e:
155            self.logger.error(f"Error fetching coordinates via JSON for {tgn_uri}: {e}")
156            return (None, None)
157
158    def get_best_match(self, results: dict, place_name: str, fuzzy_threshold: float) -> tuple:
159        if not results:
160            return (None, None)
161        
162        if len(results) == 1:
163            return self.get_coordinates_lod_json(results[0].get("p", {}).get("value", ""))
164
165        for r in results:
166            label = r.get("pLab", {}).get("value", "")
167            uri = r.get("p", {}).get("value", "")
168            ratio = fuzz.ratio(label.lower(), place_name.lower())
169            if ratio >= fuzzy_threshold:
170                self.logger.info(f"Best match for '{place_name}': {label} ({ratio}%)")
171                return self.get_coordinates_lod_json(uri)
172        
173        return (None, None)

175class WHGQuery(BaseQuery):
176    """
177    A class to interact with the World Historical Gazetteer (WHG) API.
178
179    This class provides methods to search and retrieve geographic coordinates for historical
180    places using the WHG API. It supports filtering by country code and feature class,
181    and includes functionality to find the best matching place from multiple results.
182
183    Attributes:
184        endpoint (str): The base URL for the WHG API
185        search_domain (str): The API endpoint path for searches. Default is "/index"
186        collection (str): The WHG collection to search in (default: "")
187
188    Example:
189        >>> whg = WHGQuery("https://whgazetteer.org/api")
190        >>> results = whg.places_by_name("Cuicatln", country_code="MX", place_type="p")
191        >>> coordinates = whg.get_best_match(results, place_type="pueblo", country_code="MX")
192    """
193    def __init__(self, search_domain: str = "index", dataset: str = ""):
194        super().__init__(base_url=config["apis"]["whg_endpoint"])
195        self.dataset = dataset
196        self.search_domain = search_domain
197
198    @sleep_and_retry
199    @limits(calls=5, period=1)  # There's no official rate limit for WHG, but we set a conservative limit
200    def places_by_name(self, place_name: str, country_code: str, place_type: str = "p") -> dict:
201        """
202        Search for place using the World Historical Gazetteer API https://docs.whgazetteer.org/content/400-Technical.html#api
203        
204        Parameters:
205            place_name (str): Any string with the name of the place. This keyword includes place names variants.
206            country_code (str): ISO 3166-1 alpha-2 country code.
207            place_type (str): Feature class according to Linked Places Format. Default is 'p' for place. Look at https://github.com/LinkedPasts/linked-places-format for more places classes.
208        """
209        
210        if not place_name or not isinstance(place_name, str):
211            raise ValueError("place_name must be a non-empty string")
212        if country_code and (not isinstance(country_code, str) or len(country_code) != 2):
213            raise ValueError("country_code must be a valid 2-letter country code")
214        if not place_type:
215            self.logger.warning("place_type should be a string, defaulting to 'p' for place type.")
216            place_type = "p"
217
218        url = f"{self.base_url}/{self.search_domain}/?name={place_name}&ccodes={country_code}&fclass={place_type}&dataset={self.dataset}"
219
220        try:
221            response = self._limited_get(url)
222            results = response.json()
223            return self._post_filtering(results, country_code=country_code)
224        except requests.exceptions.RequestException as e:
225            self.logger.error(f"Request error searching for '{place_name}': {str(e)}")
226            return {"features": []}
227        except ValueError as e:
228            self.logger.error(f"Invalid JSON response for '{place_name}': {str(e)}")
229            return {"features": []}

232    def get_best_match(self, results: dict, place_name: str, fuzzy_threshold: float) -> tuple:
233
234        self.logger.info(f"Finding best match for '{place_name}' in WHG results")
235
236        try:
237            features = results.get("features", [])
238            if not features:
239                return None, None
240
241            for r in features:
242                name = r.get("properties", {}).get("title", "")
243                if not name:
244                    continue
245                
246                ratio = fuzz.ratio(name.lower(), place_name.lower())
247                self.logger.info(f"Comparing '{name}' with '{place_name}': {ratio}% similarity")
248                if ratio >= fuzzy_threshold:
249                    geometry = r.get("geometry", {})
250                    if geometry.get("type") == "GeometryCollection":
251                        self.logger.warning(f"Best match for '{place_name}' is a GeometryCollection. Taking the first valid point.")
252
253                        coordinates = None
254                        for geom in geometry.get("geometries", []):
255                            if geom.get("type") == "Point":
256                                coordinates = geom.get("coordinates")
257                                break
258                        if not coordinates:
259                            self.logger.warning(f"No valid Point found in GeometryCollection for '{place_name}'.")
260                            continue
261                        
262                    else:
263                        coordinates = geometry.get("coordinates")
264                    if coordinates and len(coordinates) == 2:
265                        self.logger.info(f"Best match for '{place_name}': {name} ({ratio}%)")
266                        return coordinates[1], coordinates[0] # Convert from GeoJSON (lon, lat) to (lat, lon)
267
268            return (None, None)
269        
270        except Exception as e:
271            self.logger.error(f"Error processing results: {str(e)}")
272            return (None, None)

274    def _post_filtering(
275    self,
276    results: dict,
277    country_code: Optional[str] = None
278) -> dict:
279        """
280        Post-process the WHG API results to filter by country code. This extra step is necessary
281        because the WHG API does a soft filtering by country code, but it does not guarantee that
282        all results will match the provided country code.
283        """
284        if not results.get("features"):
285            return {"features": []}
286
287        filtered = []
288        for feature in results["features"]:
289            props = feature.get("properties", {})
290            ccodes = props.get("ccodes", [])
291            if len(ccodes) == 0:
292                ccodes = feature.get("ccodes", [])
293
294            # Check country code
295            if country_code and country_code.upper() not in ccodes:
296                continue
297
298            filtered.append(feature)
299
300        return {"features": filtered}

303    """
304    A class to interact with the Geonames API.
305
306    This class provides methods to search and retrieve geographic coordinates for places
307    using the Geonames API. It supports filtering by country and feature class.
308
309    Attributes:
310        endpoint (str): The base URL for the Geonames API
311        username (str): Geonames API username for authentication
312
313    Example:
314        >>> geonames = GeonamesQuery("http://api.geonames.org", username="your_username")
315        >>> results = geonames.places_by_name("Madrid", country="ES")
316        >>> coordinates = geonames.get_best_match(results, "Madrid")
317    """
318    def __init__(self, geonames_username: Union[str, None] = None):
319        super().__init__(base_url=config["apis"]["geonames_endpoint"])
320        if geonames_username:
321            self.username = geonames_username
322        else:

327    def places_by_name(self, place_name: str, country_code: str, place_type: Union[str, None] = None) -> dict:
328        """
329        Search for places using the Geonames API.
330        
331        Parameters:
332            place_name (str): Name of the place to search for
333            country_code (str): Optional ISO 3166-1 alpha-2 country code
334            place_type (str): Optional feature class (A: country, P: city/village, etc.).
335                              Additional types can be added in the data/mappings/geonames_place_map.json file.
336        """
337
338        params = {
339            'q': place_name,
340            'username': self.username,
341            'maxRows': 10,
342            'type': 'json',
343            'style': 'FULL'
344        }
345        
346        if country_code:
347            params['country'] = country_code
348        
349        if place_type:
350            params['featureClass'] = place_type.lower()
351
352        try:
353            response = self._limited_get(
354                "/searchJSON",
355                params=params
356            )
357            return response.json()
358        except Exception as e:
359            self.logger.error(f"Error querying Geonames for '{place_name}': {str(e)}")
360            return {"geonames": []}

362    def get_best_match(self, results: dict, place_name: str, fuzzy_threshold: float) -> tuple:
363        """
364        Get the best matching place from the results based on name similarity.
365        
366        Parameters:
367            results (dict): Results from places_by_name query
368            place_name (str): Original place name to match against
369            fuzzy_threshold (float): Minimum similarity score (0-100) for a match
370        
371        Returns:
372            tuple: (latitude, longitude) or (None, None) if no match found
373        """
374        if not results.get("geonames"):
375            return (None, None)
376
377        geonames = results["geonames"]
378        if len(geonames) == 1:
379            return (float(geonames[0]["lat"]), float(geonames[0]["lng"]))
380
381        best_ratio = 0
382        best_coords = (None, None)
383        
384        for place in geonames:
385            name = place.get("name", "")
386            alternate_names = place.get("alternateNames", [])
387            all_names = [name] + [n.get("name", "") for n in alternate_names]
388            
389            for n in all_names:
390                partial_ratio = fuzz.partial_ratio(place_name.lower(), n.lower())
391                regular_ratio = fuzz.ratio(place_name.lower(), n.lower())
392                ratio = max(partial_ratio, regular_ratio)
393                
394                if ratio > best_ratio:
395                    best_ratio = ratio
396                    best_coords = (float(place["lat"]), float(place["lng"]))
397                    self.logger.info(f"Found match: '{name}' with similarity {ratio}%")
398
399        if best_ratio >= fuzzy_threshold:
400            return best_coords
401        
402        return (None, None)

404class WikidataQuery(BaseQuery):
405    """
406    A class to interact with the Wikidata MediaWiki API for geographic coordinates lookup.
407    """
408
409    def __init__(self,
410                 search_endpoint="https://www.wikidata.org/w/api.php",
411                 entitydata_endpoint="https://www.wikidata.org/wiki/Special:EntityData/"):
412        super().__init__(base_url=search_endpoint)
413        self.search_endpoint = search_endpoint
414        self.entitydata_endpoint = entitydata_endpoint
415
416    @sleep_and_retry
417    @limits(calls=30, period=1)
418    def places_by_name(self, place_name: str, country_code: str, place_type: Union[str, None] = None) -> list:
419        params = {
420            "action": "wbsearchentities",
421            "search": place_name,
422            "language": "en",
423            "format": "json",
424            "type": "item",
425            "limit": 10
426        }
427
428        try:
429            response = self._limited_get(self.search_endpoint, params=params)
430            search_results = response.json().get("search", [])
431        except Exception as e:
432            self.logger.error(f"Error querying Wikidata for '{place_name}': {e}")
433            return []
434
435        enriched_results = []
436
437        for result in search_results:
438            qid = result.get("id")
439            label = result.get("label", "")
440
441            # Fetch entity data
442            entity_data = self._fetch_entity_data(qid)
443            if not entity_data:
444                continue
445
446            claims = entity_data.get("claims", {})
447            coords = self._extract_coordinates(claims)
448            if not coords or coords == (None, None):
449                continue
450
451            if country_code and not self._match_country(claims, country_code):
452                continue
453
454            if place_type and not self._match_place_type(claims, place_type):
455                continue
456
457            enriched_results.append({
458                "label": label,
459                "qid": qid,
460                "coordinates": coords
461            })
462
463        return enriched_results
464
465    def _fetch_entity_data(self, qid: str) -> dict:
466        try:
467            url = f"{self.entitydata_endpoint}{qid}.json"
468            response = self._limited_get(url)
469            return response.json()["entities"][qid]
470        except Exception as e:
471            self.logger.warning(f"Failed to fetch entity data for {qid}: {e}")
472            return {}

474    def get_best_match(self, results: dict, place_name: str, fuzzy_threshold: float = 90) -> tuple:
475        if not results:
476            return (None, None)
477
478        best_score = 0
479        best_coords = None
480
481        for result in results:
482            label = result["label"]
483            coords = result["coordinates"]
484            score = max(fuzz.ratio(label.lower(), place_name.lower()),
485                        fuzz.partial_ratio(label.lower(), place_name.lower()))
486
487            if score > best_score and score >= fuzzy_threshold:
488                best_score = score
489                best_coords = coords
490                self.logger.info(f"Wikidata match: '{label}'  {score}%")
491
492        return best_coords if best_coords else (None, None)
493
494    def _extract_coordinates(self, claims: dict) -> tuple:
495        try:
496            coord_data = claims.get("P625", [])[0]["mainsnak"]["datavalue"]["value"]
497            return coord_data["latitude"], coord_data["longitude"]
498        except Exception:
499            return (None, None)
500
501    def _match_country(self, claims: dict, iso_code: str) -> bool:
502        try:
503            country_entity = claims.get("P17", [])[0]["mainsnak"]["datavalue"]["value"]["id"]
504            url = f"{self.entitydata_endpoint}{country_entity}.json"
505            response = self._limited_get(url)
506            country_data = response.json()
507            wikidata_iso = country_data["entities"][country_entity]["claims"]["P297"][0]["mainsnak"]["datavalue"]["value"]
508            return wikidata_iso.upper() == iso_code.upper()
509        except Exception:
510            return False
511
512    def _match_place_type(self, claims: dict, expected_qid: str) -> bool:
513        try:
514            types = [c["mainsnak"]["datavalue"]["value"]["id"] for c in claims.get("P31", [])]
515            return expected_qid in types
516        except Exception:
517            return False

520class PlaceResolver:
521    """
522    A unified resolver that queries multiple geolocation services in order
523    and returns the first match with valid coordinates.
524    """
525    def __init__(self, services: Optional[List[BaseQuery]] = None, places_map_json: str = "data/mappings/places_map.json",
526                 threshold: float = 90, verbose: bool = False):
527        
528        if services is None or not isinstance(services, list) or len(services) == 0:
529            services = [
530                GeonamesQuery(),
531                WHGQuery(),
532                TGNQuery(),
533                WikidataQuery()
534            ]
535
536        self.services = services
537        self.places_map = self._load_places_map(places_map_json)
538        self.threshold = threshold
539        self.logger = setup_logger(self.__class__.__name__, verbose)
540
541        for service in self.services:
542            service.logger = setup_logger(service.__class__.__name__, verbose)
543            self.logger.debug(f"Updated logger for {service.__class__.__name__} with verbose={verbose}")
544
545    def _load_places_map(self, json_file: str) -> dict:
546        try:
547            with open(json_file, "r") as f:
548                return json.load(f)
549        except Exception as e:
550            self.logger.error(f"Error loading places map: {e}")
551            raise ValueError(f"Could not load places map from {json_file}. Ensure the file exists and is valid JSON.")
552
553    def resolve(self, place_name: str, country_code: Union[str, None] = None, place_type: Union[str, None] = None,
554               use_default_filter: bool = False) -> tuple:
555        """
556        Try resolving the place coordinates using multiple sources.
557
558        Args:
559            place_name (str): The place name to search
560            country_code (str): ISO country code (optional)
561            place_type (str): Place type (optional)
562            use_default_filter (bool): If True, apply a default filter as fallback in case the place_type is not found.
563                                        If no place_type is provided, no filtering will be applied.
564
565        Returns:
566            tuple: (lat, lon) or (None, None) if not found
567        """
568        for service in self.services:
569            try:
570                self.logger.info(f"Trying {service.__class__.__name__} for '{place_name}'")
571                mapper = PlaceTypeMapper(self.places_map)
572                service_key = service.__class__.__name__.lower().replace("query", "")
573
574                resolved_type = None
575
576                if place_type:
577                    resolved_type = mapper.get_for_service(place_type, service_key)
578                    if resolved_type is None and use_default_filter:
579                        self.logger.warning(
580                            f"Unrecognized place_type '{place_type}' for service '{service_key}', falling back to 'pueblo'."
581                        )
582                        resolved_type = mapper.get_for_service("pueblo", service_key)
583                    elif resolved_type is None:
584                        self.logger.info(
585                            f"Skipping place_type filter for service '{service_key}' (unrecognized type: '{place_type}')."
586                        )
587
588                results = service.places_by_name(place_name, country_code, resolved_type)
589                coords = service.get_best_match(results, place_name, fuzzy_threshold=self.threshold)
590                if coords != (None, None):
